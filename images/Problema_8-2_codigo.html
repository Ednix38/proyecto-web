        let num;
        num = prompt('Ingrese un valor entero');

            document.write('El número ingresado es cero');
        } else {
            if (num > 0) {
                document.write('El número ingresado es positivo');
            } else {
                document.write('El número ingresado es negativo');
            }
        }
    </script>
